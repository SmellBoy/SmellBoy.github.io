x = linspace(0,40,100);

T = linspace(0,pi,100);

y1 = 1 + sin(T)/2 + sin(T*9)/3/2 + 0.1*T;

y2 = 1 + sin(T)/2 + sin(T*9)/4/2 + 0.1*T;

y3 = 1 + sin(T)/2 + sin(T*9)/5/2 + 0.1*T;

plot(x,y1,x,y2,x,y3);

axis tight;

grid on;

xlabel('上课时间/分钟');

ylabel('心率百分比/%');
